//app.js
app.run("[demoApp]", ["config"], function (config) {
    //mandatory step
    config.templateFunc = function (templateName, obj) {
        //templates is pre-compiled handlerbar templates object
        return templates[templateName](obj);
    }
});
app.component("helloComponent", function(){
    this.render("homeTemplate");
});
